# K            : K    = s/(g*3.6*3.6)   Sabit, hat açıklığına bağlı                    #
# MİNİMUM YARIÇAP : Rmin = K             * (Vmax * Vmax )/(hi+hmax)                        #
#                                                                                       #
# HER BİR R YARIÇAP DEĞERİ İÇİN:                                                          #
#               TEORİK DEVER    : ht   = K * (Vmax*Vmax)/R                                 #
#                            : htr  = ht - hi;                                          #
#               GERÇEK DEVER       : hr   = Eğer(hmax < htr) ise->hmax  Değilse->htr                   #
#               DEVER KANUNU: hp   = Eğer(hmax < htr) ise->hmax  Değilse->hmax*Rmin/R           #
#                    Kapli>0 : hk   = Kapli*K*(Vmax*Vmax)/R                             #
#               PRATİK DEVER   : hp   = Eğer(hmax < hk) ise->hmax  Değilse->hk                     #
#                                                                                       #
#                Eğer Kapli değeri sıfırdan farklı ise pratik veya uygulanan dever #
#                açıklanan şekilde hesaplanır, aksi takdirde oransal bir dever   #
#                kanunu izlenir.                                                  #
#                                                                                       #
#               V maksimum(R)  : Vm   = Eğer(hmax < htr) ise->Kök((hr+hi)*R/K)  Değilse->Vmax     #
#               GERÇEK Hi      : hir  = K*(Vm*Vm)/R - hp                                  #
#               GERÇEK anc     : ancr = hir*g/s                                           #
#                                                                                       #
# R İÇİN MİNİMUM KLOTOİD UZUNLUĞU HESABI:                                  #
#                                                                                       #
#               Geometrik Sınırlama    : LminRampa =  hp/Rampa                        #
#               Konfor Sınırlaması       : LminVhi   =  hir*Vm/(Vhi*3,6)                #
#               Dinamik Sınırlama      : LminVh    =  hp*Vm/(Vh*3,6)                  #
#               Klotoid Uzunluğu         : L         =  EnBüyük(LminRampa,LminVhi,LminVh) #
#               Klotoid A               : A         =  Kök(R*L)                       #
# MİNİMUM HIZ           : Vmin = Eğer(hp>he) ise->3,6*Kök(g*R/s*(hp-he)) Değilse->0        #
