#  1/8   Maksimum dever geçiş uzunluğunun analizi.              #
#  2/8   Drenaj kontrolü, çatı eğiminden düşük deverli bölgelerin sınırı    #
#  3/8   Verilen bir değerden düşük eğime sahip düşey güzergah kesimleri hakkında uyar    #
#  4/8   Klotoitsiz kesimlerde dever geçişini kontrol et             #
#  5/8   S kurpları arasındaki dever geçişlerinin analizi               #
#  6/8   C kurplarında dever geçişi (kısa aliymanlarda çatı eğimini önler)  #
#  7/8   Çatı eğimine geçişe eşlik etme (Otoyollar ve Ekspres Yollar)          #
#  8/8   Otomatik uzatmalı tek kaplamalı kollar için deverler     #
#                                                                              #
####### ---------------------------------------------------------------------- #
# 1/8 #  Minimum dever geçiş uzunluğunun analizi               #
####### ---------------------------------------------------------------------- #
# Maksimum göreli kenar eğimi, ulaşılabilecek deveri sınırlar  #
# eğer klotoit kısaysa.  dever = uzunluk * ipmax / genişlik                     #
# -----   ---     ------------------------------------------------------------ #
<GRT1-2>
# ipmax   0.6     # ipmax = 1.8 - 0.01 * Vp (proje hızı 120 için 0.6)   #
</GRT1-2>
<GRT3>
# ipmax   1.0     # ipmax = 1.8 - 0.01 * Vp (proje hızı 80 için 1.0)    #
</GRT3>
####### ---------------------------------------------------------------------- #
# 2/8 #  Çatı eğiminden daha az deverli kesimlerin maksimum sınırı               #
####### ---------------------------------------------------------------------- #
# Düşey güzergah eğimi < mras ise, maksimum çatı eğimi geçiş uzunluğu.                #
# Kurp-aliyman / klotoit-aliyman çatı eğimi geçişini belirtir         #
# Ayrıca, büyük klotoitler durumunda, suyun birikebileceği kesimleri önlemek #
# için çatı eğimine eşit bir dever verisi eklenir.                    #
#                                                                              #
# Eğer çatı eğiminden daha az deverli kesim Mlbo'dan daha uzunsa, #
# Lbo metre mesafeye çatı eğimi değerinde bir dever verisi eklenir.                            #
#                                                                              #
# Sadece düşey güzergah eğimi mras'tan düşükse analiz edilir.             #
# Eğer mras = 0.0 ise, düşey güzergahtan bağımsız olarak her zaman kontrol eder.                         #
#                                                                              #
#       Mlbo  mras  Lbo                                                        #
#------ ----  ----  ---- ----------------------------------------------------- #
<GRT1-2>
 Klmbom 40.0  0.00  40.0                                                       #
</GTR1-2>
<GRT3>
 Klmbom 20.0  0.00  20.0                                                       #
</GRT3>
#                                                                              #
####### ---------------------------------------------------------------------- #
# 3/8 #  Düşük deverli bölgelerde yetersiz eğime sahip alanlar hakkında bilgilendir #
#######  Minimum_Düşey_Güzergah_Eğimi %  Dever <= % olan bölgelerde                  #
#------- --------------------------  ------------------------ ---------------- #

# 6/8 #  C kurplarında dever geçişi (kısa aliymanlarda çatı eğimini önler)  #
####### ---------------------------------------------------------------------- #
# Belirli bir değerden daha kısa uzunluktaki aliymanlar için, çatı eğimi yerine, #
# kurpta çatı eğimi değerinde ve kurpların dönüş yönünde bir deverli kesim bırakılır. #
#                                                                              #
# Bu devere geçiş, klotoitin 'çatı eğimi'ne karşılık gelen yarıçap değerine    #
# sahip olduğu noktadan itibaren otomatik olarak gerçekleştirilir.      #
#                                                                              #
# İki klotoitten biri (veya her ikisi) eksikse, Rmtr uzunluğunda bir geçiş          #
# denenir; eğer kalan aliymanın uzunluğu kontrol değerinden küçükse,      #
# çatı eğimine geçiş yapılmaz ve dever, ilgili noktada çatı eğimi   #
# değerinde kalır.                       #
#                                                                              #
# C Kurplarında (Ovoidler) Çatı Eğiminden Kaçın   Maks. Aliyman Uzun.                     #
